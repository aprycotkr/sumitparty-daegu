// 검토중 상태 전원에게 리마인드 일괄 발송
function sendBulkReminder() {

  const ss    = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const sheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (!sheet) { ui.alert('신청자 시트를 찾을 수 없습니다.'); return; }

  const data  = sheet.getDataRange().getValues();
  let count = 0;
  for (let i = 1; i < data.length; i++) {
    const status = String(data[i][COL.STATUS] || '').trim();
    if (status !== '검토중') continue;
    if (!data[i][COL.NAME] || !data[i][COL.PHONE]) continue;
    sendReminderSMS(sheet, i + 1);
    count++;
    Utilities.sleep(300);
  }
  ui.alert('리마인드 발송 완료: ' + count + '명에게 발송되었습니다.');
}

// ── 진입점 ────────────────────────────────────────────────────
function doGet(e) {
  const action = e.parameter.action || '';
  try {
    if (action === 'getDateStatus') {
      return jsonResponse(getDateStatus());
    }
    if (action === 'updateDateStatus') {
      return jsonResponse(updateDateStatusFromGet(e.parameter));
    }
    if (action === 'submitApplication') {
      const payload = JSON.parse(e.parameter.data || '{}');
      return jsonResponse(submitApplication(payload));
    }
    return jsonResponse({ result: 'error', message: 'unknown action' });
  } catch (err) {
    return jsonResponse({ result: 'error', message: err.message });
  }
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const action = payload.action || '';
    if (action === 'submitApplication') {
      return jsonResponse(submitApplication(payload));
    }
    if (action === 'updateDateStatus') {
      return jsonResponse(updateDateStatus(payload));
    }
    if (action === 'submitGuest') {
      const guestPayload = {
        action:  'submitApplication',
        name:    payload.name,
        gender:  payload.gender,
        phone:   payload.phone,
        date:    payload.date,
        price:   payload.price,
        ticket:  payload.ticket || '게스트',
        refBy:   payload.refBy  || '직접등록',
      };
      return jsonResponse(submitApplication(guestPayload));
    }
    return jsonResponse({ result: 'error', message: 'unknown action' });
  } catch (err) {
    return jsonResponse({ result: 'error', message: err.message });
  }
}

// ── 날짜 상태 조회 ────────────────────────────────────────────
// 날짜관리 시트 구조:
//   A: 날짜 (YYYY-MM-DD)  B: 남성상태  C: 여성상태  D: 남성잔여(수동)  E: 여성잔여(수동)
//   상태값: open / almost / closed
// D/E에 값이 있으면 수동 설정값 우선, 없으면 신청자 시트 자동 집계
function getDateStatus() {
  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const data = {};

  // 날짜관리 시트에서 상태 + 수동 잔여석 읽기
  const dateSheet = ss.getSheetByName(CONFIG.DATE_SHEET_NAME);
  if (dateSheet) {
    dateSheet.getDataRange().getValues().forEach(function(row) {
      const dateKey = toDateKey(row[0]);
      if (!dateKey || !/^\d{4}-\d{2}-\d{2}$/.test(dateKey)) return;
      const entry = { male: row[1] || 'open', female: row[2] || 'open' };
      // D컬럼(row[3]), E컬럼(row[4]) — 관리자가 직접 입력한 잔여석
      const mL = row[3];
      const fL = row[4];
      if (mL !== '' && mL !== null && mL !== undefined) {
        entry.maleRemaining = Number(mL);
        entry.hasManualRemaining = true;
      }
      if (fL !== '' && fL !== null && fL !== undefined) {
        entry.femaleRemaining = Number(fL);
        entry.hasManualRemaining = true;
      }
      data[dateKey] = entry;
    });
  }

  // 신청자 시트에서 날짜별 남녀 인원 집계 (불합격 제외)
  // 수동 설정값이 없는 경우에만 자동 집계값 사용
  const applySheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (applySheet && applySheet.getLastRow() > 1) {
    const rows = applySheet.getRange(2, 1, applySheet.getLastRow() - 1, 11).getValues();
    const counts = {};
    rows.forEach(function(row) {
      const dateKey = toDateKey(row[COL.DATE]);
      const gender  = String(row[COL.GENDER] || '').trim();
      const status  = String(row[COL.STATUS] || '').trim();
      if (!dateKey || !gender || status.startsWith('불합격')) return;
      if (!counts[dateKey]) counts[dateKey] = { male: 0, female: 0 };
      if (gender === '남성') counts[dateKey].male++;
      else if (gender === '여성') counts[dateKey].female++;
    });
    Object.keys(counts).forEach(function(key) {
      if (!data[key]) data[key] = { male: 'open', female: 'open' };
      if (data[key].maleRemaining === undefined)
        data[key].maleRemaining   = Math.max(0, 20 - counts[key].male);
      if (data[key].femaleRemaining === undefined)
        data[key].femaleRemaining = Math.max(0, 20 - counts[key].female);
    });
  }

  return { result: 'success', data: data };
}

// ── 날짜 상태 업데이트 (GET 방식 — 어드민에서 호출) ──────────
function updateDateStatusFromGet(params) {
  const dateStr    = String(params.date || '').trim();
  if (!dateStr) return { result: 'error', message: 'date required' };
  const male       = params.male       || 'open';
  const female     = params.female     || 'open';
  const maleLeft   = params.maleLeft   !== undefined ? params.maleLeft   : '';
  const femaleLeft = params.femaleLeft !== undefined ? params.femaleLeft : '';
  return updateDateStatus({ updates: [{ date: dateStr, male, female, maleLeft, femaleLeft }] });
}

// ── 날짜 상태 업데이트 ────────────────────────────────────────
// payload: { action, updates: [{ date: 'YYYY-MM-DD', male: 'open', female: 'open' }, ...] }
// 날짜관리 시트에 행이 없으면 추가, 있으면 남성상태/여성상태 덮어씀
function updateDateStatus(payload) {
  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  let sheet = ss.getSheetByName(CONFIG.DATE_SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.DATE_SHEET_NAME);
    sheet.appendRow(['날짜', '남성상태', '여성상태']);
    sheet.setFrozenRows(1);
  }

  const updates = payload.updates || [];

  updates.forEach(function(u) {
    const dateStr    = String(u.date || '').trim();
    if (!dateStr) return;
    const male       = u.male       || 'open';
    const female     = u.female     || 'open';
    const maleLeft   = u.maleLeft   !== undefined ? u.maleLeft   : '';
    const femaleLeft = u.femaleLeft !== undefined ? u.femaleLeft : '';

    const allData = sheet.getDataRange().getValues();
    let found = false;
    for (let i = 1; i < allData.length; i++) {
      const cell = allData[i][0];
      let cellKey = '';
      if (cell instanceof Date) {
        cellKey = Utilities.formatDate(cell, 'Asia/Seoul', 'yyyy-MM-dd');
      } else {
        cellKey = String(cell || '').trim();
      }
      if (cellKey === dateStr) {
        sheet.getRange(i + 1, 2, 1, 4).setValues([[male, female, maleLeft, femaleLeft]]);
        found = true;
        break;
      }
    }
    if (!found) {
      sheet.appendRow([dateStr, male, female, maleLeft, femaleLeft]);
    }
  });

  return { result: 'success' };
}

// ── 신청서 제출 ───────────────────────────────────────────────
// payload: { action, date, name, gender, bdate, insta, phone, ticket, price, photoUrl, branch }
function submitApplication(payload) {
  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  let sheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.SHEET_NAME);
    sheet.appendRow(['제출시간','파티날짜','이름','성별','생년월일','인스타그램','연락처','권종','금액','사진URL','심사결과']);
    sheet.setFrozenRows(1);
  }

  // 중복 신청 방지: 같은 날짜 + 같은 연락처
  const existing = sheet.getDataRange().getValues();
  for (let i = 1; i < existing.length; i++) {
    if (
      String(existing[i][COL.PHONE]) === String(payload.phone) &&
      toDateKey(existing[i][COL.DATE]) === String(payload.date)
    ) {
      return { result: 'duplicate', message: '이미 신청된 연락처입니다.' };
    }
  }

  const now = new Date();
  const row = new Array(11).fill('');
  row[COL.TIMESTAMP]  = Utilities.formatDate(now, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  row[COL.DATE]       = payload.date || '';
  row[COL.NAME]       = payload.name || '';
  row[COL.GENDER]     = payload.gender || '';
  row[COL.BIRTH]      = payload.birth || payload.bdate || '';
  row[COL.INSTAGRAM]  = payload.instagram || payload.insta || '';
  row[COL.PHONE]      = payload.phone || '';
  row[COL.TICKET]     = payload.ticket || '1+2부';
  row[COL.PRICE]      = payload.price || '';
  row[COL.PHOTO_URL]  = payload.photoUrl || '';
  row[COL.STATUS]     = '검토중';

  sheet.appendRow(row);

  // 신청 즉시 입금 안내 SMS 발송
  const phone = String(payload.phone || '').replace(/[^0-9]/g, '');
  if (phone && payload.name) {
    const priceNum = payload.price ? Number(payload.price).toLocaleString() : '';
    sendSms(phone, buildApplyMsg(payload.name, payload.date || '', priceNum));
  }

  // Meta CAPI: 서버사이드 Lead 이벤트 전송
  sendMetaCAPI(payload);

  return { result: 'success' };
}

// ── 심사결과 변경 트리거 ──────────────────────────────────────
// 스프레드시트에서 K열(인덱스 10)을 수동으로 "합격" 또는 "불합격"으로 바꾸면
// onEdit 트리거가 실행되어 자동으로 SMS 발송.
//
// 설치: Apps Script 편집기 → 트리거 추가 → onEdit → 스프레드시트 편집 시
function onEdit(e) {
  try {
    const sheet = e.range.getSheet();
    if (sheet.getName() !== CONFIG.SHEET_NAME) return;

    const col = e.range.getColumn();      // 1-based
    const row = e.range.getRow();
    if (col !== COL.STATUS + 1) return;   // K열 = 11번째
    if (row <= 1) return;                 // 헤더 제외

    const newStatus = String(e.value || '').trim();
    if (newStatus !== '합격' && newStatus !== '불합격' && newStatus !== '리마인드') return;

    if (newStatus === '합격')    { sendApprovalSMS(sheet, row); }
    else if (newStatus === '리마인드') { sendReminderSMS(sheet, row); }
    else if (newStatus === '불합격')  { sendRejectionSMS(sheet, row); }
  } catch (err) {
    console.error('onEdit error:', err);
  }
}

// ── 심사결과별 SMS 함수 ───────────────────────────────────────

function sendApprovalSMS(sheet, row) {
  const rowData = sheet.getRange(row, 1, 1, 11).getValues()[0];
  const phone   = String(rowData[COL.PHONE] || '').replace(/\D/g, '');
  const name    = rowData[COL.NAME]   || '';
  const gender  = rowData[COL.GENDER] || '';
  const price   = rowData[COL.PRICE]  ? Number(rowData[COL.PRICE]).toLocaleString() : '';
  const rawDate = rowData[COL.DATE];
  if (!phone || !name) return;

  const dateObj = rawDate instanceof Date ? rawDate : new Date(rawDate);
  const dow     = ['일','월','화','수','목','금','토'][dateObj.getDay()];
  const dateStr = (dateObj.getMonth()+1) + '/' + dateObj.getDate() + '(' + dow + ')';

  sendSms(phone, buildPassMsg(name, gender, dateStr, price));
  sheet.getRange(row, COL.STATUS + 1).setValue('합격✓발송');

  // 합격 = 입금 확인 완료 → Purchase CAPI 전송
  const priceNum = rowData[COL.PRICE] ? parseInt(String(rowData[COL.PRICE]).replace(/[^0-9]/g, '')) || 0 : 0;
  sendMetaCAPIPurchase({ phone: phone, name: name, gender: gender, price: priceNum });

  updateRosterSheet();
}

// ── Meta CAPI Purchase 전송 ───────────────────────────────────
function sendMetaCAPIPurchase(data) {
  try {
    function sha256(value) {
      if (!value) return null;
      const bytes = Utilities.computeDigest(
        Utilities.DigestAlgorithm.SHA_256,
        value.toString().trim().toLowerCase()
      );
      return bytes.map(function(b){ return ('0' + (b & 0xFF).toString(16)).slice(-2); }).join('');
    }
    function normalizePhone(phone) {
      let p = String(phone).replace(/\D/g, '');
      if (p.startsWith('0')) p = '82' + p.slice(1);
      return p;
    }

    const genderCode = data.gender === '여성' ? 'f' : 'm';
    const ud = {
      ph: data.phone ? sha256(normalizePhone(data.phone)) : null,
      fn: (data.name && data.name.length > 1) ? sha256(data.name.slice(1)) : null,
      ln: data.name ? sha256(data.name.slice(0, 1)) : null,
      ge: sha256(genderCode)
    };
    Object.keys(ud).forEach(function(k){ if (ud[k] === null) delete ud[k]; });

    const payload = {
      data: [{
        event_name:       'Purchase',
        event_time:       Math.floor(new Date().getTime() / 1000),
        event_id:         'daegu_purchase_' + Date.now(),
        action_source:    'website',
        event_source_url: 'https://daegu.sumitparty.com/daegu-apply.html',
        user_data:        ud,
        custom_data:      { value: data.price || 0, currency: 'KRW' }
      }]
    };

    UrlFetchApp.fetch(
      'https://graph.facebook.com/v21.0/' + PIXEL_ID + '/events?access_token=' + CAPI_ACCESS_TOKEN,
      { method: 'POST', contentType: 'application/json', payload: JSON.stringify(payload), muteHttpExceptions: true }
    );
  } catch(err) {}
}

// ── Solapi SMS 발송 ───────────────────────────────────────────
function sendSms(to, text) {
  if (!CONFIG.SOLAPI_API_KEY || !CONFIG.SOLAPI_API_SECRET) {
    console.warn('Solapi 키 미설정 — SMS 미발송:', to);
    return;
  }

  const date   = new Date().toISOString();
  const salt   = Utilities.getUuid();
  const hmac   = buildHmac(date, salt);

  const payload = {
    message: {
      to:   to,
      from: CONFIG.SENDER_NUMBER,
      text: text,
    }
  };

  const options = {
    method:      'post',
    contentType: 'application/json',
    headers: {
      Authorization: 'HMAC-SHA256 apiKey=' + CONFIG.SOLAPI_API_KEY +
                     ', date=' + date +
                     ', salt=' + salt +
                     ', signature=' + hmac,
    },
    payload:           JSON.stringify(payload),
    muteHttpExceptions: true,
  };

  const res = UrlFetchApp.fetch('https://api.solapi.com/messages/v4/send', options);
  const json = JSON.parse(res.getContentText());
  if (json.errorCode) {
    console.error('SMS 발송 실패:', json.errorCode, json.errorMessage);
  }
}

function buildHmac(date, salt) {
  const msg  = date + salt;
  const key  = Utilities.newBlob(CONFIG.SOLAPI_API_SECRET).getBytes();
  const data = Utilities.newBlob(msg).getBytes();
  const sig  = Utilities.computeHmacSignature(Utilities.MacAlgorithm.HMAC_SHA_256, data, key);
  return sig.map(function(b){ return ('0' + (b & 0xff).toString(16)).slice(-2); }).join('');
}

// ── 날짜 셀 값 → yyyy-MM-dd 변환 ────────────────────────────
// 스프레드시트가 날짜 문자열을 Date 객체로 자동 변환하므로 통일 처리
function toDateKey(val) {
  if (!val) return '';
  if (val instanceof Date) return Utilities.formatDate(val, 'Asia/Seoul', 'yyyy-MM-dd');
  return String(val).trim();
}

// ── 합격자 명단 자동 정리 (시트1) ────────────────────────────
// 날짜별 고정 28행 구조 (getLastRow 대신 nextRow로 직접 위치 추적):
//   +0  날짜 헤더 수식  |  E열: __DATE__:yyyy-MM-dd 마커
//   +1  열 헤더 [이름|연락처|이름|연락처]
//   +2~+26  25슬롯 — 합격자 위에서 자동 채움, 빈 칸에 직접 입력 가능
//   +27 빈 구분 행
function updateRosterSheet() {
  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const applySheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (!applySheet) return;

  const rosterSheet = ss.getSheetByName('시트1') || ss.getSheets()[0];
  if (!rosterSheet || rosterSheet.getName() === CONFIG.SHEET_NAME) return;

  const MAX = 25;
  const SEC = 2 + MAX + 1; // 섹션 1개 = 헤더(1) + 열헤더(1) + 데이터(25) + 빈행(1) = 28
  const DOW = ['일','월','화','수','목','금','토'];

  // 합격자 날짜별 수집
  const applyData = applySheet.getDataRange().getValues();
  const dateMap = {};
  for (let i = 1; i < applyData.length; i++) {
    const row    = applyData[i];
    const status = String(row[COL.STATUS] || '').trim();
    if (status.indexOf('합격') === -1) continue;
    const dateKey = toDateKey(row[COL.DATE]);
    if (!dateKey) continue;
    if (!dateMap[dateKey]) dateMap[dateKey] = { male: [], female: [] };
    const gender = String(row[COL.GENDER] || '').trim();
    const entry  = [String(row[COL.NAME] || ''), String(row[COL.PHONE] || '')];
    if (gender === '남성') dateMap[dateKey].male.push(entry);
    else                   dateMap[dateKey].female.push(entry);
  }

  // 기존 날짜 섹션 탐색 (E열 __DATE__: 마커 기반)
  const sheetLastRow = rosterSheet.getLastRow();
  const sectionMap = {};
  if (sheetLastRow > 0) {
    const markerVals = rosterSheet.getRange(1, 5, sheetLastRow, 1).getValues();
    for (let r = 0; r < markerVals.length; r++) {
      const m = String(markerVals[r][0] || '');
      if (m.startsWith('__DATE__:')) sectionMap[m.slice(9)] = r + 1;
    }
  }

  // 열 너비 최초 1회 설정
  const props = PropertiesService.getScriptProperties();
  if (!props.getProperty('rosterColsInit')) {
    rosterSheet.setColumnWidth(1, 130);
    rosterSheet.setColumnWidth(2, 175);
    rosterSheet.setColumnWidth(3, 130);
    rosterSheet.setColumnWidth(4, 175);
    rosterSheet.hideColumns(5);
    props.setProperty('rosterColsInit', '1');
  }

  // 새 섹션 시작 위치를 getLastRow 대신 직접 계산
  // (빈 문자열 setValues는 getLastRow에 반영 안 됨)
  let nextRow = 1;
  Object.keys(sectionMap).forEach(function(key) {
    nextRow = Math.max(nextRow, sectionMap[key] + SEC);
  });

  // 날짜별 섹션 생성/업데이트
  Object.keys(dateMap).sort().forEach(function(dateStr) {
    const male   = dateMap[dateStr].male.slice(0, MAX);
    const female = dateMap[dateStr].female.slice(0, MAX);
    const d      = new Date(dateStr + 'T00:00:00+09:00');
    const label  = (d.getMonth()+1) + '월 ' + d.getDate() + '일 (' + DOW[d.getDay()] + ')';

    let startRow = sectionMap[dateStr];

    if (!startRow) {
      startRow = nextRow;
      nextRow  = startRow + SEC;
      sectionMap[dateStr] = startRow;

      // 날짜 헤더
      rosterSheet.getRange(startRow, 1, 1, 4).merge()
        .setBackground('#1a1a1a').setFontColor('#C9A96E')
        .setFontWeight('bold').setFontSize(12).setHorizontalAlignment('center');
      rosterSheet.getRange(startRow, 5).setValue('__DATE__:' + dateStr);

      // 열 헤더
      rosterSheet.getRange(startRow + 1, 1, 1, 2)
        .setValues([['이름', '연락처']])
        .setBackground('#0d1020').setFontColor('#8899ff').setFontWeight('bold').setFontSize(10);
      rosterSheet.getRange(startRow + 1, 3, 1, 2)
        .setValues([['이름', '연락처']])
        .setBackground('#200d0d').setFontColor('#ff8888').setFontWeight('bold').setFontSize(10);

      // 25행 빈 슬롯 미리 생성
      const emptyRows = Array.from({length: MAX}, function(){ return ['','','','']; });
      rosterSheet.getRange(startRow + 2, 1, MAX, 4).setValues(emptyRows);
    }

    // 합격자 + 수동 추가 인원 병합 후 기록
    const autoStart = startRow + 2;

    // 기존 슬롯 데이터 읽기
    const existing = rosterSheet.getRange(autoStart, 1, MAX, 4).getValues();

    // 합격자 key set 구성
    const maleKeySet   = {};
    const femaleKeySet = {};
    male.forEach(function(m)   { if (m[0]) maleKeySet[m[0]   + '|' + m[1]] = true; });
    female.forEach(function(f) { if (f[0]) femaleKeySet[f[0] + '|' + f[1]] = true; });

    // 수동 추가 인원 추출 (합격자 목록에 없는 사람)
    const manualM = [], manualF = [];
    existing.forEach(function(r) {
      const mn = String(r[0] || '').trim(), mp = String(r[1] || '').trim();
      const fn = String(r[2] || '').trim(), fp = String(r[3] || '').trim();
      if (mn && !maleKeySet[mn + '|' + mp])   manualM.push([mn, mp]);
      if (fn && !femaleKeySet[fn + '|' + fp]) manualF.push([fn, fp]);
    });

    // 합격자 먼저, 수동 추가 뒤에 배치
    const finalM = male.concat(manualM).slice(0, MAX);
    const finalF = female.concat(manualF).slice(0, MAX);

    // MAX 전체 행 덮어쓰기 (빈 슬롯 포함)
    const fullRows = [];
    for (let i = 0; i < MAX; i++) {
      fullRows.push([
        (finalM[i] || ['', ''])[0], (finalM[i] || ['', ''])[1],
        (finalF[i] || ['', ''])[0], (finalF[i] || ['', ''])[1]
      ]);
    }
    rosterSheet.getRange(autoStart, 1, MAX, 4).setValues(fullRows);

    // 날짜 헤더 수식 (해당 섹션 25행만 카운트)
    const dataEnd = autoStart + MAX - 1;
    rosterSheet.getRange(startRow, 1).setFormula(
      '="📅 ' + label + '  남성 "&COUNTA(A' + autoStart + ':A' + dataEnd + ')&"/' + MAX + '명  ·  여성 "&COUNTA(C' + autoStart + ':C' + dataEnd + ')&"/' + MAX + '명"'
    );
  });
}

// ── Meta CAPI Lead 전송 ───────────────────────────────────────
function sendMetaCAPI(payload) {
  try {
    function sha256(value) {
      if (!value) return null;
      const bytes = Utilities.computeDigest(
        Utilities.DigestAlgorithm.SHA_256,
        value.toString().trim().toLowerCase()
      );
      return bytes.map(function(b){ return ('0' + (b & 0xFF).toString(16)).slice(-2); }).join('');
    }
    function normalizePhone(phone) {
      let p = phone.replace(/\D/g, '');
      if (p.startsWith('0')) p = '82' + p.slice(1);
      return p;
    }

    const genderCode = payload.gender === '여성' ? 'f' : 'm';
    const birth = String(payload.birth || payload.bdate || '').replace(/[^0-9]/g, '');
    const dob = birth.length >= 8 ? birth.slice(0, 8) : null;

    const ud = {
      ph:  payload.phone ? sha256(normalizePhone(String(payload.phone))) : null,
      fn:  (payload.name && payload.name.length > 1) ? sha256(payload.name.slice(1)) : null,
      ln:  payload.name  ? sha256(payload.name.slice(0, 1)) : null,
      ge:  sha256(genderCode),
      db:  dob ? sha256(dob) : null,
      fbc: payload.fbc || null,
      fbp: payload.fbp || null,
      client_user_agent: payload.userAgent || null
    };
    Object.keys(ud).forEach(function(k){ if (ud[k] === null) delete ud[k]; });

    const apiPayload = {
      data: [{
        event_name:       'Lead',
        event_time:       Math.floor(new Date().getTime() / 1000),
        event_id:         payload.eventId || ('daegu_gas_' + Date.now()),
        action_source:    'website',
        event_source_url: 'https://daegu.sumitparty.com/daegu-apply.html',
        user_data:        ud
      }]
    };

    UrlFetchApp.fetch(
      'https://graph.facebook.com/v21.0/' + PIXEL_ID + '/events?access_token=' + CAPI_ACCESS_TOKEN,
      {
        method:             'POST',
        contentType:        'application/json',
        payload:            JSON.stringify(apiPayload),
        muteHttpExceptions: true
      }
    );
  } catch(err) {
    // 조용히 실패 — 신청 처리는 막지 않음
  }
}
